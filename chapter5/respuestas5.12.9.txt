4. Realiza un programa que calcule el área y perímetro del cuadrado
    #include <stdio.h>

    int main() {
        float lado, area, perimetro;

        printf("Introduce el valor del lado del cuadrado: ");
        scanf("%f", &lado);

        area = lado * lado;
        perimetro = 4 * lado;

        printf("Area del cuadrado: %f\n", area);
        printf("Perimetro del cuadrado: %f\n", perimetro);

        return 0;
    }

5. Realiza un programa que calcule el área y perímetro del rectángulo
    #include <stdio.h>

    int main() {

        float base, altura, area, perimetro;

        printf("Introduce la base del rectangulo: ");
        scanf("%f", &base);

        printf("Introduce la altura del rectangulo: ");
        scanf("%f", &altura);

        area = base * altura;
        perimetro = 2 * (base + altura);

        printf("Area del rectangulo: %f\n", area);
        printf("Perimetro del rectangulo: %f\n", perimetro);

        return 0;
    }
